#Analisis del ciclo de paso, hasta el instante donde impacta el talon con el suuelo, utilizando algoritmo ZUPT para forzar las condiciones v=0 en
#el instante final.

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy.signal import butter, filtfilt
from scipy.spatial.transform import Rotation as R

# 1. Cargar datos
acc_df = pd.read_csv('Accelerometer.csv')
gyr_df = pd.read_csv('Gyroscope.csv')

# 2. Filtrar por tiempo
t0, t1 = 5.30, 6.34
acc_df = acc_df[(acc_df['seconds_elapsed'] >= t0) & (acc_df['seconds_elapsed'] <= t1)].reset_index(drop=True)
gyr_df = gyr_df[(gyr_df['seconds_elapsed'] >= t0) & (gyr_df['seconds_elapsed'] <= t1)].reset_index(drop=True)

# 3. Obtener tiempo
acc_t = acc_df['seconds_elapsed'].values
gyr_t = gyr_df['seconds_elapsed'].values

# 4. Interpolar giroscopio al tiempo del acelermetro
gyro_interp = np.vstack([
    np.interp(acc_t, gyr_t, gyr_df['x']),
    np.interp(acc_t, gyr_t, gyr_df['y']),
    np.interp(acc_t, gyr_t, gyr_df['z'])
]).T

# 5. Filtrado pasa-bajo Butterworth
fs = 1.0 / np.mean(np.diff(acc_t))
b, a = butter(4, 6 / (fs / 2), btype='low')
acc_filt = filtfilt(b, a, acc_df[['x', 'y', 'z']].values, axis=0)

# 6. Corregir bias inicial (primeros 0.2 s)
bias_samples = int(0.2 * fs)
bias = np.mean(acc_filt[:bias_samples], axis=0)
acc_corr = acc_filt - bias

# 7. Integrar orientacin (quaternion acumulando rotvecs)
rots = [R.identity()]
for i in range(1, len(acc_t)):
    dt = acc_t[i] - acc_t[i - 1]
    omega = gyro_interp[i] * dt
    rotvec = R.from_rotvec(omega)
    rots.append(rots[-1] * rotvec)

rot_matrices = np.stack([r.as_matrix() for r in rots])

# 8. Rotar aceleraciones al marco global
acc_global = np.einsum('nij,nj->ni', rot_matrices, acc_corr)

# 9. Integrar aceleracin  velocidad (RK4)
def integrate_rk4(time, signal):
    v = np.zeros_like(signal)
    for i in range(1, len(time)):
        dt = time[i] - time[i-1]
        k1 = signal[i-1]
        k2 = 0.5 * (signal[i-1] + signal[i])
        k3 = k2
        k4 = signal[i]
        v[i] = v[i-1] + (dt/6.0) * (k1 + 2*k2 + 2*k3 + k4)
    return v

vel_global = integrate_rk4(acc_t, acc_global)

# 10. Aplicar ZUPT: forzar velocidad final = 0
zupt_corr = np.zeros_like(vel_global)
for i in range(3):
    drift = np.linspace(0, vel_global[-1, i], len(acc_t))
    zupt_corr[:, i] = vel_global[:, i] - drift

# 11. Integrar velocidad corregida  posicin
pos_global = integrate_rk4(acc_t, zupt_corr)

# 12. Graficar posiciones y velocidades
for i, axis in enumerate(['X', 'Y', 'Z']):
    plt.figure()
    plt.plot(acc_t, zupt_corr[:, i], label=f'Velocidad {axis}')
    plt.plot(acc_t, pos_global[:, i], label=f'Posicin {axis}')
    plt.xlabel('Tiempo (s)')
    plt.ylabel('Valor (m / m/s)')
    plt.title(f'Velocidad y Posicin en eje {axis}')
    plt.legend()
    plt.grid()
    plt.tight_layout()
    plt.show()

# 13. Trayectoria lateral (X vs Y)
plt.figure(figsize=(8, 6))
plt.plot(pos_global[:, 0], pos_global[:, 1], marker='o', label='Trayectoria lateral')
plt.xlabel('Desplazamiento X (m)')
plt.ylabel('Altura Y (m)')
plt.title('Trayectoria del tobillo en vista lateral (Y vs X)')
plt.grid()
plt.legend()
plt.tight_layout()
plt.show()
